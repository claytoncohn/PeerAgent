[When Green Flag Clicked]
[Set x_position to (-60)]
[Set x_velocity to (0)]
[Set x_acceleration to (4)]
[Start Simulation]

[Simulation Step]
[Change x_position by (DeltaT)(*)(x_velocity)]
[Change x_velocity by (DeltaT)(*)(x_acceleration)]
[If (x_velocity)(>)(15)]
    [Set x_velocity to (15)]
[If (x_position)(>)(8)]
    [Set x_acceleration to (-4)]
[If (x_position)(>)(37.95)]
    [Set x_velocity to (0)]
    [Stop Simulation]
